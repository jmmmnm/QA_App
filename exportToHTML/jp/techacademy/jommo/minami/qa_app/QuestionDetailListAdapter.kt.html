13   class QuestionDetailListAdapter(context: Context, private val mQustion: Question) : BaseAdapter() {

15           private val TYPE_QUESTION = 0 
16           private val TYPE_ANSWER = 1 
17       } 
18    
19       private var mLayoutInflater: LayoutInflater? = null 
20    
21       init { 
22           mLayoutInflater = context.getSystemService(Context.LAYOUT_INFLATER_SERVICE) as LayoutInflater 
23       } 
24    
25       override fun getCount(): Int { 
26           return 1 + mQustion.answers.size 
27       } 
28    
29       override fun getItemViewType(position: Int): Int { 
30           return if (position == 0) { 
31               TYPE_QUESTION 
32           } else { 
33               TYPE_ANSWER 
34           } 
35       } 
36    
37       override fun getViewTypeCount(): Int { 
38           return 2 
39       } 
40    
41       override fun getItem(position: Int): Any { 
42           return mQustion 
43       } 
44    
45       override fun getItemId(position: Int): Long { 
46           return 0 
47       } 
48    
49       override fun getView(position: Int, convertView: View?, parent: ViewGroup): View { 
50           var convertView = convertView 
51    
52           if (getItemViewType(position) == TYPE_QUESTION) { 
53               if (convertView == null) { 
54                   convertView = mLayoutInflater!!.inflate(R.layout.list_question_detail, parent, false)!! 
55               } 
56               val body = mQustion.body 
57               val name = mQustion.name 
58    
59               val bodyTextView = convertView.findViewById<View>(R.id.bodyTextView) as TextView 
60               bodyTextView.text = body 
61    
62               val nameTextView = convertView.findViewById<View>(R.id.nameTextView) as TextView 
63               nameTextView.text = name 
64    
65               val bytes = mQustion.imageBytes 
66               if (bytes.isNotEmpty()) { 
67                   val image = BitmapFactory.decodeByteArray(bytes, 0, bytes.size).copy(Bitmap.Config.ARGB_8888, true) 
68                   val imageView = convertView.findViewById<View>(R.id.imageView) as ImageView 
69                   imageView.setImageBitmap(image) 
70               } 
71           } else { 
72               if (convertView == null) { 
73                   convertView = mLayoutInflater!!.inflate(R.layout.list_answer, parent, false)!! 
74               } 
75    
76               val answer = mQustion.answers[position - 1] 
77               val body = answer.body 
78               val name = answer.name 
79    
80               val bodyTextView = convertView.findViewById<View>(R.id.bodyTextView) as TextView 
81               bodyTextView.text = body 
82    
83               val nameTextView = convertView.findViewById<View>(R.id.nameTextView) as TextView 
84               nameTextView.text = name 
85           } 
86    
87           return convertView 
88       }
