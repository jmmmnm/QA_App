20           mLayoutInflater = context.getSystemService(Context.LAYOUT_INFLATER_SERVICE) as LayoutInflater 
21       }

24           return mQuestionArrayList.size 
25       }

27       override fun getItem(position: Int): Any {

29       }

33       } 
34    
35       override fun getView(position: Int, convertView: View?, parent: ViewGroup): View { 
36           var convertView = convertView 
37    
38           if (convertView == null) { 
39               convertView = mLayoutInflater.inflate(R.layout.list_questions, parent, false) 
40           } 
41    
42           val titleText = convertView!!.findViewById<View>(R.id.titleTextView) as TextView 
43           titleText.text = mQuestionArrayList[position].title 
44    
45           val nameText = convertView.findViewById<View>(R.id.nameTextView) as TextView 
46           nameText.text = mQuestionArrayList[position].name 
47    
48           val resText = convertView.findViewById<View>(R.id.resTextView) as TextView 
49           val resNum = mQuestionArrayList[position].answers.size 
50           resText.text = resNum.toString() 
51    
52           val bytes = mQuestionArrayList[position].imageBytes 
53           if (bytes.isNotEmpty()) { 
54               val image = BitmapFactory.decodeByteArray(bytes, 0, bytes.size).copy(Bitmap.Config.ARGB_8888, true) 
55               val imageView = convertView.findViewById<View>(R.id.imageView) as ImageView 
56               imageView.setImageBitmap(image) 
57           } 
58    
59           return convertView 
60       }
